555  template <class Predicate>
556  int32 removeAll(const Predicate& predicate)
557  {
558  const int32 originalNum = _arrayNum;
559  if (!originalNum)
560  {
561  return 0; // nothing to do, loop assumes one item so need to deal with this edge case here
562  }
563 
564  int32 writeIndex = 0;
565  int32 readIndex = 0;
566  bool notMatch = !predicate(getData()[readIndex]); // use a ! to guarantee it can't be anything other than zero or one
567  do
568  {
569  int32 runStartIndex = readIndex++;
570  while (readIndex < originalNum && notMatch == !predicate(getData()[readIndex]))
571  {
572  readIndex++;
573  }
574  int32 runLength = readIndex - runStartIndex;
575  AE_ASSERT(runLength > 0);
576  if (notMatch)
577  {
578  // this was a non-matching run, we need to move it
579  if (writeIndex != runStartIndex)
580  {
581  Memory::memmove(&getData()[writeIndex], &getData()[runStartIndex], sizeof(ElementType)* runLength);
582  }
583  writeIndex += runLength;
584  }
585  else
586  {
587  // this was a matching run, delete it
588  Memory::destructItems(getData() + runStartIndex, runLength);
589  }
590  notMatch = !notMatch;
591  } while (readIndex < originalNum);
592 
593  _arrayNum = writeIndex;
594  return originalNum - _arrayNum;
595  }
