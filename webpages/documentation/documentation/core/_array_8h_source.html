562  template <class Predicate>
563  int32 removeAll(const Predicate& predicate)
564  {
565  const int32 originalNum = _arrayNum;
566  if (!originalNum)
567  {
568  return 0; // nothing to do, loop assumes one item so need to deal with this edge case here
569  }
570 
571  int32 writeIndex = 0;
572  int32 readIndex = 0;
573  bool notMatch = !predicate(getData()[readIndex]); // use a ! to guarantee it can't be anything other than zero or one
574  do
575  {
576  int32 runStartIndex = readIndex++;
577  while (readIndex < originalNum && notMatch == !predicate(getData()[readIndex]))
578  {
579  readIndex++;
580  }
581  int32 runLength = readIndex - runStartIndex;
582  AE_ASSERT(runLength > 0);
583  if (notMatch)
584  {
585  // this was a non-matching run, we need to move it
586  if (writeIndex != runStartIndex)
587  {
588  Memory::memmove(&getData()[writeIndex], &getData()[runStartIndex], sizeof(T)* runLength);
589  }
590  writeIndex += runLength;
591  }
592  else
593  {
594  // this was a matching run, delete it
595  Memory::destructItems(getData() + runStartIndex, runLength);
596  }
597  notMatch = !notMatch;
598  } while (readIndex < originalNum);
599 
600  _arrayNum = writeIndex;
601  return originalNum - _arrayNum;
602  }
